# Usage

Just like `useAnyOther` hook, only better:


```jsx
const RequestForm = (props) => {
  const {
    value: name, 
    bind: bindName, 
    isValid: isNameValid
  } = useInput({
    value: 'John',
    name: 'request-form-name'
    validateOnChange: true,
  })

  const {
    value: address, 
    bind: bindAddress, 
    isValid: isAddressValid 
  } = useInput({
    placeholder: '1234 Street Road',
    name: 'request-form-address'
    validateOnChange: true,
    validOnDefault: false,
  })

  return(
    <div>
      <input {...bindName} />
      <input {...bindAddress} />
      <button disabled={!isNameValid || !isAddressValid}>
        Submit
      </button>
      <p>{name} @ {address}</p>
    </div>
  )
}
```
