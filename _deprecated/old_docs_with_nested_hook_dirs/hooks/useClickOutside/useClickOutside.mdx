## Usage

<Tabs>

<TabItem value='simple' label='Simple'>

```tsx
import { useClickOutside } from '@pratiq/hooks'

const ref = useClickOutside(()=>{
    // callback invoked on click outside event
})

<p ref={ref}>Click Here! (or not)</p>
```

</TabItem>

<TabItem value='advanced' label='Advanced'>

```tsx 
import { useClickOutside } from '@pratiq/hooks'

const [isMenuOpen, setIsMenuOpen] = useState(false)

const ref = useClickOutside(()=>{
    setIsMenuOpen(false)
})

<button onClick={()=> setIsMenuOpen(true)}>Open Menu</button>
<div id='menu' ref={ref}></div>
```

</TabItem>

<TabItem value='full' label='Full'>

```tsx 
import { useClickOutside } from '@pratiq/hooks'

const [isMenuOpen, setIsMenuOpen] = useState(false)
const customRef = useRef(null)

useClickOutside(()=>{
    setIsMenuOpen(false)
}, customRef)

<button onClick={()=> setIsMenuOpen(true)}>Open Menu</button>

<div id='menu' ref={customRef}>
    <a href='#'>Link</a>
    <a href='#'>Link</a>
    <a href='#'>Link</a>
</div>

```

</TabItem>

## API 


### Config Example

```ts
const myRef = useRef(null)
cosnt clickOut = useClickOutside(() => {}, myRef)
```




### Interfaces

```ts
type T_UseClickOutside = (callback: (e:React.MouseEvent) => void) => (el:any) => void;
```

## Hook Methods

<Card>

### clickOut
Register a react ref on an element using the ref prop

```ts
() => void 
```
```tsx
cosnt clickOut = useClickOutside(() => {}, myRef)

<div ref={clickOut}>Click outside of this div...</div>
<div ref={clickOut}>...or this one!</div>
```

</Card>

## Internal

Registers an event listener on the provided ref, or creates a new ref to attach 
listeners to.
