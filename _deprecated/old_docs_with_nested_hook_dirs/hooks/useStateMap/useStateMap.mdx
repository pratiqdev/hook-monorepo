## Usage    
 
<Tabs>

<TabItem value='simple' label='Simple'>

```js 
import { useStateMap } from '@pratiq/hooks'

const myMap = useStateMap( [ ['one', 1], ['two', 2] ] )

myMap.set('three', 3)
```

</TabItem>

<TabItem value='advanced' label='Advanced'>

```js
import { useStateMap } from '@pratiq/hooks'

const { value, set, get, remove, clear, reset } = useStateMap( [ ['one', 1], ['two', 2] ] )

set('three', 3) // set a value
get('two')      // get a value
value           // get the map


<p>{get('three')}</p>
<pre>{JSON.stringify( Array.from( value.entries() ) )}</pre>
```

</TabItem>
