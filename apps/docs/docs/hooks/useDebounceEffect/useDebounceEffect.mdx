## Usage

<Tabs>

<TabItem value='simple' label='Simple'>

```js
useDebounceEffect(()=>{
    // calback invoked on any state change
    // default debounce time is 250ms
})
```

</TabItem>

<TabItem value='advanced' label='Advanced'>

```js
useDebounceEffect(()=>{
    // calback invoked once on mount
    // default debounce time is 250ms
}, [])
```

</TabItem>


<TabItem value='full' label='Full'>

```js
useDebounceEffect(()=>{
    // calback invoked once on mount
    // and on dep change
    // debounce time is 1s
}, [dep], 1000)
```

</TabItem>

</Tabs>
