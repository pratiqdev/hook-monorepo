## Usage

<Tabs>

<TabItem value='simple' label='Simple'>

```tsx live
function Eventful() {

    useEventListener('click', (e) => {
        // do work
    })

    return (
        <div>
            
        </div>
    )
}
```

</TabItem>

<TabItem value='advanced' label='Advanced'>

```tsx live
function Eventful(){

    const elementRef = useRef<any>(null)
    const [isEnabled, toggle] = useEventListener('click', (e) => {
        console.log('CLICK:', e)
    }, elementRef.current)

    return(
        <>
            <Button ref={elementRef}>
                Click Me!
            </Button>

            <Button ref={elementRef}>
                Click Me!
            </Button>

            <Button onClick={toggle}>
                Toggle
            </Button>
        </>
    )
}
```

</TabItem>

</Tabs>

## API 

Values used for initial state and internal settings are provided as a config
object 

### Config Example

```ts
const [isEnabled, toggle] = useEventListener('click', () => {
     // ... 
}, myRef)
```



### Return Values

<HeadlessTable items={[
    ['@isEnabled', 'boolean', 'True if the eventListener is currently registered and active'],
]}/>

### Interfaces

```ts
type T_UseEventListener = (event:string, callback:Function, element: any) => [
    isEnabled:boolean,
    toggle: () => void
]
```
